wijmo/wijmo.angular2.chart.hierarchical

Contains Angular 2 components for the wijmo.chart.hierarchical module.

wijmo.angular2.chart.hierarchical is an external TypeScript module that can be imported to your code using its ambient module name. For example:

import * as wjHierarchical from 'wijmo/wijmo.angular2.chart.hierarchical';
 
@Component({
    directives: [wjHierarchical.WjSunburst],
    template: `
      <wj-sunburst [itemsSource]="data" [binding]="'y'" [bindingX]="'x'">
      </wj-sunburst>`,
    selector: 'my-cmp',
})
export class MyCmp {
    data: any[];
}
