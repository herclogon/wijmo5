wijmo/wijmo.angular2.chart.radar

Contains Angular 2 components for the wijmo.chart.radar module.

wijmo.angular2.chart.radar is an external TypeScript module that can be imported to your code using its ambient module name. For example:

import * as wjRadar from 'wijmo/wijmo.angular2.chart.radar';
 
@Component({
    directives: [wjRadar.WjFlexRadar, wjRadar.WjFlexRadarSeries],
    template: `
      <wj-flex-radar [itemsSource]="data" [bindingX]="'x'">
          <wj-flex-radar-series [binding]="'y'"></wj-flex-radar-series>
      </wj-flex-radar>`,
    selector: 'my-cmp',
})
export class MyCmp {
    data: any[];
}
